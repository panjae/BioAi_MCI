# Week 6-2: MS (Mind Scanner) Module class MindScanner: def __init__(self): self.cognitive_scores = { 'memory': [], 'attention': [], 'language': [], 'executive': [] } def memory_test(self): # Word list memory test words = random.sample(["apple", "desk", "sky", "music", "happiness", "travel", "friend", "coffee", "walk", "ocean"], 5) st.write("üß† Memory Test") st.write(f"Remember these words for 30 seconds:") cols = st.columns(5) for i, word in enumerate(words): cols[i].info(word) time.sleep(30) st.empty() # Interference task st.write("Let's talk about something else briefly.") distractor = st.text_input("How's the weather today?") # Recall test recalled = st.text_input("Please write all the words you saw earlier:") score = self.calculate_memory_score(words, recalled) self.cognitive_scores['memory'].append(score) return score def attention_test(self): # Reverse digit span test st.write("üéØ Attention Test") difficulty = len(self.cognitive_scores['attention']) + 3 numbers = [random.randint(1, 9) for _ in range(difficulty)] st.write(f"Say these numbers in reverse: {' '.join(map(str, numbers))}") answer = st.text_input("Answer:") correct = ' '.join(map(str, reversed(numbers))) score = 100 if answer.replace(' ', '') == correct.replace(' ', '') else 0 self.cognitive_scores['attention'].append(score) return score def generate_report(self): # Generate cognitive function assessment report report = { 'overall_score': np.mean([ np.mean(scores) for scores in self.cognitive_scores.values() if scores ]), 'memory_trend': self.calculate_trend(self.cognitive_scores['memory']), 'attention_trend': self.calculate_trend(self.cognitive_scores['attention']), 'recommendation': self.get_recommendation() } return report
